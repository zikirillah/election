  334.                 
  335.                 $xml .= '            <Cell '.$style.' ss:Index="'.$column.'" '.$URL.' '.$mergeCell.' '.$formula.'>'."\r";

  337.                 $xml .= $data;
  338.                 $xml .= '</Data>'."\r";

  341.             }
  342.             $xml .= '        </Row>'."\r";
  343.         }
